function doGet() { const ss = SpreadsheetApp.getActiveSpreadsheet(); // ── 1. 조직도 시트에서 이름 + 직급 읽기 ── const orgSheet = ss.getSheetByName("조직도"); const orgData = orgSheet.getDataRange().getValues(); const rankMap = {}; const ranks = ['대표','부대표','부장','차장','과장','대리','사원','인턴','연구원','책임연구원','센터장','공장장','이사','CSO','COO','팀장']; for (let i = 0; i < orgData.length; i++) { const row = orgData[i]; for (let j = 0; j < row.length - 1; j++) { const cell = String(row[j] || '').trim(); const next = String(row[j+1] || '').trim(); if (cell && next && ranks.some(r => next.includes(r))) { rankMap[cell] = next; } } } // ── 2. R&R 시트에서 연락처 + R&R 읽기 ── const rrSheet = ss.getSheetByName("3. 비상연락망 및 R&R"); const rrData = rrSheet.getDataRange().getValues(); const result = []; let currentDept = ''; let currentDeptRR = ''; for (let i = 1; i < rrData.length; i++) { const row = rrData[i]; const no = row[1]; // B열 - 번호 const name = String(row[2] || '').trim(); // C열 - 이름 const office = String(row[3] || '').trim(); // D열 - 사무실 const ext = String(row[4] || '').trim(); // E열 - 내선 const phone = String(row[5] || '').trim(); // F열 - 핸드폰 const dept = String(row[6] || '').trim(); // G열 - 부서 const rr = String(row[7] || '').trim(); // H열 - R&R // 빈 행 건너뜀 if (!name) continue; // 부서명 업데이트 (G열에 값 있을 때) if (dept) { currentDept = dept; currentDeptRR = ''; } // ★ 핵심: B열(No.)이 숫자인 행만 사람으로 처리 const noNum = Number(no); if (!no || isNaN(noNum) || noNum <= 0) { // 번호 없는 행 = 그룹메일/팀명 헤더 행 → R&R만 저장 if (rr) currentDeptRR = rr; continue; } // 개인 R&R 없으면 부서 공통 R&R 사용 const finalRR = rr || currentDeptRR; const rank = rankMap[name] || ''; result.push({ no: noNum, name: name, rank: rank, office: office, ext: ext, phone: phone, dept: currentDept, rr: finalRR }); } return ContentService .createTextOutput(JSON.stringify(result)) .setMimeType(ContentService.MimeType.JSON); }
